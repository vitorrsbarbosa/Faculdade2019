DECLARE 
	CURSOR C_AULA IS SELECT RA, MEDIA, FALTAS/CARGA_HORA PERC_FALTAS, RESULTADO
	FOR UPDATE OF RESULTADO;
	
	
BEGIN
	OPEN C_AULA
	FOR RC_AULA.RA IN C_AULA LOOP
	IF RC_AULA.MEDIA < 70 THEN
		UPDATE ALUNO SET RESULTADO = 'REPROVADO'
	END LOOP;
	FOR RC_AULA.RA IN C_AULA LOOP
	IF RC.AULA.PERC_FALTAS > 25
		UPDATE ALUNO SET RESULTADO = 'REPROVADO'
	END LOOP
	.
	.
	.
